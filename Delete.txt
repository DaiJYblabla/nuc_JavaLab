class ListNode{
	int val;
	ListNode next;
	public ListNode(int x){
		val = x;
	}
}
class Sol{
	ListNode head ;
	public ListNode removeElements(ListNode head,int val){
		ListNode node = head;
		if(node == null){
			System.out.println("链表中没有数");
		}
		while(node.next != null){
			if(node.next.equals(val)){
				node.next = node.next.next;
			}
		}
		
		return node;
		
	}
}
